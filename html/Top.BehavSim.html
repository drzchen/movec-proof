Lemma Error__inversion : ∀ R,
  Error R →
  R = ROK ∨ R = RAbort ∨ R = RSystemError.
Proof.
  intros R H.
  destruct R.
    unfold Error in H. destruct H as [H | H]. inversion H. inversion H.
    unfold Error in H. destruct H as [H | H]. inversion H. inversion H.
    inversion H. inversion H0.
    inversion H0.
    auto.
    auto.
Qed.
